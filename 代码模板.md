### 二分法代码模板

```javascript
const bSearch = (a: number[], n, value) => {
  let low = 0
  let high = n - 1

  while (low <= high) { 
    const mid = (low + high) / 2; 
    if (a[mid] == value) { 
      return mid
    } else if (a[mid] < value) {
      low = mid + 1
    } else { 
      high = mid - 1
    }
  }
  return -1
}
```

### 并查集

```javascript
// 一维模板
class UnionFind {
  constructor(n) {
    this.graph = [...Array(n)].map((_, i) => i)
    this.groups = n
  }

  groups: number
  
  find(id) {
    if(this.graph[id] === id) return id;
    this.graph[id] = this.find(this.graph[id]);
    return this.graph[id];
  }
  
  union(x, y) {
    const rootX = this.find(x);
    const rootY = this.find(y);
    if(rootX !== rootY) {
        this.graph[rootY] = rootX
        this.groups--
    }
  }
}
```
