### 递归代码模板

```javascript
const recursion = (level, params) => {
  // 1. 终止条件
  if (level > maxIndex) {
    process_result
    return 
  }
  // 2. 当前层逻辑处理
  process(level, params)
  // 3. 去到下一层递归
  recursion(level + 1, params)
  // 4. 可能：如果需要恢复当前参数的状态（用的比较少）
  reverse_state(level)
}
```

```javascript
const divide_conquer = (problem, param1, param2, ...) => {
  // 1. 终止条件

    process_result

  }

```
